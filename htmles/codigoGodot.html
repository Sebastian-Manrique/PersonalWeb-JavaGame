using Godot;

using System;


public class playerNewNew : KinematicBody2D

{

public const float OriginalSpeed = 200.0f;

public const float SlowerSpeed = 170.0f; // Nueva velocidad reducida


public override void _Process(float delta)

{

Vector2 input = Vector2.Zero;


if (Input.IsActionPressed("ui_right"))

input.x += 1;

if (Input.IsActionPressed("ui_left"))

input.x -= 1;

if (Input.IsActionPressed("ui_down"))

input.y += 1;

if (Input.IsActionPressed("ui_up"))

input.y -= 1;


// Normalizar el vector de entrada

if (input.Length() > 0)

input = input.Normalized();


// Usar la velocidad reducida

Vector2 velocity = input * SlowerSpeed;


// Mover el cuerpo kinem√°tico

Translate(velocity * delta);

}

}
